164 int read_sdo(ec_sdo_request_t *req)
165 {
166  int sdo_read_value;
167  switch (ecrt_sdo_request_state(req)) {
168  case EC_REQUEST_UNUSED: // request was not used yet
169  ecrt_sdo_request_read(req); // trigger first read
170  break;
171  case EC_REQUEST_BUSY:
172  //fprintf(stderr, "SDO still busy...\n");
173  break;
174  case EC_REQUEST_SUCCESS:
175  sdo_read_value = EC_READ_S32(ecrt_sdo_request_data(req));
176  //logmsg(1, "SDO value read: 0x%X\n",
177  // sdo_read_value);
178  ecrt_sdo_request_read(req); // trigger next read
179  break;
180  case EC_REQUEST_ERROR:
181  //fprintf(stderr, "Failed to read SDO!\n");
182  ecrt_sdo_request_read(req); // retry reading
183  break;
184  }
185  return sdo_read_value;
186 }

191

194  ecrt_sdo_request_write(req); // trigger first read

197  //fprintf(stderr, "SDO write still busy...\n");
198  //logmsg(1, "SDO value written: \n",data );

202  //logmsg(1, "SDO value written: 0x%X\n", data);
203  pause();
204  ecrt_sdo_request_write(req); // trigger next read ???

206  break;

374 void init_master(master_setup_variables_t *master_setup, ctrlproto_slv_handle *slv_handles, unsigned int total_no_of_slaves)
375 {
376  int slv;
377 
378  struct sigaction sa;
379  struct itimerval tv;
380 
381  master_setup->master = ecrt_request_master(0);
382  if (!master_setup->master)
383  exit(-1);
384 
385  master_setup->domain = ecrt_master_create_domain(master_setup->master);
386  if (!master_setup->domain)
387  exit(-1);
388 
389  for (slv = 0; slv < total_no_of_slaves; ++slv)
390  {
391  if (!( slv_handles[slv].slave_config= ecrt_master_slave_config( //sc_data_in
392  master_setup->master, slv_handles[slv].slave_alias, slv_handles[slv].slave_pos , slv_handles[slv].slave_vendorid, slv_handles[slv].slave_productid))) {
393  fprintf(stderr, "Failed to get slave configuration.\n");
394  exit(-1);
395  }
396 
397  //logmsg(1, "Configuring PDOs...\n");
398  if (ecrt_slave_config_pdos(slv_handles[slv].slave_config, EC_END, slv_handles[slv].__sync_info)) { //slave_0_syncs
399  fprintf(stderr, "Failed to configure PDOs.\n");
400  exit(-1);
401  }
402 
403  //#if PARAMETER_UPDATE
404  motor_config_request(slv_handles[slv].slave_config, slv_handles[slv].__request);
405  //#endif
406  }
407 
408  if (ecrt_domain_reg_pdo_entry_list(master_setup->domain, master_setup->domain_regs)) {
409  fprintf(stderr, "PDO entry registration failed!\n");
410  exit(-1);
411  }
412 
413  if (ecrt_master_set_send_interval(master_setup->master, FREQUENCY) != 0) {
414  fprintf(stderr, "failed to set send interval\n");
415  exit(-1);
416  }
417  logmsg(1, "Activating master...\n");
418  if (ecrt_master_activate(master_setup->master))
419  exit(-1);
420 
421  if (!(master_setup->domain_pd = ecrt_domain_data(master_setup->domain))) {
422  exit(-1);
423  }
424 
425 
426  pid_t pid = getpid();
427  if (setpriority(PRIO_PROCESS, pid, -19))
428  fprintf(stderr, "Warning: Failed to set priority: %s\n",
429  strerror(errno));
430 
431  sa.sa_handler = signal_handler;
432  sigemptyset(&sa.sa_mask);
433  sa.sa_flags = 0;
434  if (sigaction(SIGALRM, &sa, 0)) {
435  fprintf(stderr, "Failed to install signal handler!\n");
436  exit(-1);
437  }
438 
439  //logmsg(1, "Starting timer...\n");
440  tv.it_interval.tv_sec = 0;
441  tv.it_interval.tv_usec = 1000000 / FREQUENCY;
442  tv.it_value.tv_sec = 0;
443  tv.it_value.tv_usec = 1000;
444  if (setitimer(ITIMER_REAL, &tv, NULL)) {
445  fprintf(stderr, "Failed to start timer: %s\n", strerror(errno));
446  exit(-1);
447  }
448 
449  logmsg(0, "Started.\n");
450 
451 }
